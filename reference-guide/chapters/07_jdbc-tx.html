JDBC and Transactions

If you have javax.sql.DataSource and (or) javax.transaction.UserTransaction you can use com.gf.components.jdbc.ConnectionInInvoke and (or) com.gf.components.tx.UserTransactionInInvoke for adding java.sql.Connection and (or) javax.transaction.UserTransaction into handler's context: 

//impl of DataSourceManager
import javax.sql.DataSource;

public class DataSourceManagerImpl implements DataSourceManager {

	public DataSource getDataSource() {
		return dataSource;
	}

}

//engine
DataSourceManagerImpl dataSourceManagerImpl = ...;
Engine engine = new Engine();
engine.addToContext(dataSourceManagerImpl);
engine.putFilter(ConnectionInInvoke.class);
engine.putHandler(SomeHandler.class);


//handler
import java.sql.Connection;

@Mapping(SomeAction.class)
public class SomeHandler extends Handler<SomeAction>{
	
	@Inject
	Connection connection;

	public void invoke(SomeAction action) throws Exception {
		...
	}

}
